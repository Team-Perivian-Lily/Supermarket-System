using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Xml;
using System.Xml.Linq;
using System.Xml.XPath;

namespace XmlImport
{
    class ImportExpenses
    {
        static void Main()
        {
            XmlDocument xmload = new XmlDocument(); 
            xmload.Load(@"E:\HomeworkApps\Supermarket-System\XmlImport\Sample-Vendor-Expenses.xml");

            XmlNodeList nodes = xmload.SelectNodes("/expenses-by-month/vendor");
            foreach(XmlNode vendors in nodes)
            {
                string vendorName = vendors.Attributes["name"].Value;
                
                foreach(XmlNode expense in vendors.ChildNodes)
                {
                    
                    string month = expense.Attributes["month"].Value;
                    double expenseMonth = double.Parse(expense.InnerText);
                    Console.WriteLine(vendorName+" Month :" + month + " Expense :" + expenseMonth);
                }       
            }          
        }
    }
}
